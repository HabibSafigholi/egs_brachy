66  EGS_Float getParticleEnergy(const EGS_Particle *p, bool subtractRM=true) {
67  EGS_Float rest_mass = 0;
68  if (subtractRM && p->q != 0){
69  rest_mass = the_useful->prm;

147  void update(EB_Message message, void *particle) {
148 
149  if (message == PARTICLE_INITIALIZED) {
150  EGS_Particle *p = static_cast<EGS_Particle *>(particle);
151  scoreParticleInitialized(p);
152  } else if (message == PARTICLE_ESCAPING_SOURCE) {
153  EGS_Particle *p = static_cast<EGS_Particle *>(particle);
154  scoreParticleEscapingSource(p);
155  } else if (message == PARTICLE_ESCAPING_GEOM) {
156 
157  EGS_Particle *p = static_cast<EGS_Particle *>(particle);
158  scoreParticleEscapingGeom(p);
159  }

578  if (!geometry) {
579  egsWarning("EnergyFluenceSpectrumInVoxel: unable to find geometry named `%s` \n", gname.c_str());
580  valid = false;
581  return;
582  }
583 
584  err = input->getInput("scoring region", local_scoring_region);
585  if (err) {

587  local_scoring_region = 0;
588  }
589

591  egsWarning("EnergyFluenceSpectrumInVoxel: invalid region `scoring region` defined. Assuming region 0\n");
592  local_scoring_region = 0;
593  }
594 
595  GeomRegT geomreg(geometry, local_scoring_region);
596  scoring_region = ginfo->localToGlobal(geomreg);
