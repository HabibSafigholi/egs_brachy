47 std::string string_format(const std::string fmt, ...) {
48  int size = ((int)fmt.size()) * 2 + 50; // Use a rubric appropriate for your code
49  std::string str;
50  va_list ap;
51  while (1) { // Maximum two passes on a POSIX system...
52  str.resize(size);
53  va_start(ap, fmt);
54  int n = vsnprintf((char *)str.data(), size, fmt.c_str(), ap);
55  va_end(ap);
56  if (n > -1 && n < size) { // Everything worked
57  str.resize(n);
58  return str;
59  }
60  if (n > -1) { // Needed size returned
61  size = n + 1; // For null char
62  } else {
63  size *= 2; // Guess at a larger size (OS specific)
64  }
65  }
66  return str;
67 }

260

262  // egsnrc modes
263  out << nbins <<","<<e_min<<","<< egsnrc_mode << endl;

266  for (int i=0; i < nbins; i++) {
267

269

271  prob = egsnrc_mode == 0 ? r : r/bin_width;
272 
273  out << string_format("%.5E,%.5E\n", top, prob);
274  }
275 
276  } else if (egsnrc_mode == 2) {
277  // egs++ line spectrum
278 
279  double r, dr, norm=0;
280  vector<double> mids;
281  vector<double> probabilities;
282 
283  // get non zero bins
284  for (int i=0; i < nbins; i++) {
285  getResult(i, r, dr);
286 
287  if (r > 0) {
288  mids.push_back(e_min + (i+0.5)*bin_width);
289  probabilities.push_back(r);
290  norm += r;
291  }
292  }
293 
294  for (size_t i = 0; i < mids.size(); i++) {
295  if (i == 0) {
296  out << string_format("%d,%.5E,%d\n", mids.size(), mids[i], egsnrc_mode);
297  }
298  out << string_format("%.5E,%.5E\n", mids[i], probabilities[i]/norm);
299  }
300  }
301  out.close();
302 
303  return fname;
