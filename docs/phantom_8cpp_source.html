304 void EB_Phantom::getResult(EGS_ScoringArray *score, int ireg, string type, EGS_Float &r, EGS_Float &dr) {
305 
306  score->currentResult(ireg, r, dr);
307  dr = r > 0 ? dr/r: 1;
308 
309  // scoring arrays are already normalized to number of statistically
310  // independent histories
311 
312  // correct norm from per history to per effective history
313  double eff_hist_norm = (double)effective_histories/cur_history;
314 
315  //egsInformation("Normalization: effective_histories %d, cur_history %d, eff_hist_norm %f\n",effective_histories, cur_history, eff_hist_norm);
316 
317 
318  if (type == "tlen") {
319  r *= 1.602E-10*nsources*dose_scale/eff_hist_norm;
320  } else if (type == "edep") {
321  r *= 1.602E-10*nsources*dose_scale/eff_hist_norm;
322  } else { // scatter norm
323  r *= (cur_history/total_radiant_e)*nsources*dose_scale/eff_hist_norm;
324  }
325 
326 
327 }

362 void EB_Phantom::outputDoseStats(EGS_ScoringArray *score, string type) {
363 
364  EGS_Float maxr=0;
365  int nreg = score->regions();
366 
367  for (int i=0; i < nreg; i++) {
368  EGS_Float r, dr;
369  getResult(score, i, type, r, dr);
370 
371  maxr = max(maxr, r);
372 
373  }
374 
375  EGS_Float avg=0, avg_err=0, avg_gt_20=0, avg_gt_20_err=0;
376  int n_gt_20 = 0;
377 
378  for (int i=0; i < nreg; i++) {
379  EGS_Float r, dr;
380  getResult(score, i, type, r, dr);
381  avg += r;
382  avg_err += dr;
383 
384  if (r >= maxr*0.20) {
385  n_gt_20++;
386  avg_gt_20 += r;
387  avg_gt_20_err += dr;
388  }
389  }
390 
391  string units = "Gy/hist";
392  if (type == "pr" || type == "ss" || type == "ms" || type == "to") {
393  units = "(Gy/hist)/R";
394  }
395 
396  egsInformation(" Dose Scaling = %.3G\n", dose_scale);
397  egsInformation(" Average dose = %.3G %s\n", avg/nreg, units.c_str());
398  egsInformation(" Average error on dose = %.3G %\n", 100*avg_err/nreg);
399  egsInformation(" Average error on dose > 0.2*Dmax = %.3G %\n", 100*avg_gt_20_err/n_gt_20);
400  egsInformation(" # of voxels with dose > 0.2*Dmax = %d (%.3G%)\n", n_gt_20, 100.*double(n_gt_20)/nreg);
401

443  );
444  } else {
445  egsInformation(

449
