445 int EB_Application::checkSourceOverlaps(EGS_Input *inp) {
446 
447  timing_blocks.addTimer("egs_brachy::checkSourceOverlaps");
448 
449  vector<string> yn_choices;
450  yn_choices.push_back("no");
451  yn_choices.push_back("yes");
452  bool check_source_overlap = (bool)inp->getInput("check source overlaps", yn_choices, 0);
453 
454  if (!check_source_overlap || source_transforms.size() <= 1){
455  /* not requested or only 1 source so no need to check for overlap */
456  timing_blocks.stopTimer();
457  return 0;
458  }
459 
460  vector<string> mode_choices;
461  mode_choices.push_back("warning");
462  mode_choices.push_back("fatal");
463 
464  bool fatal = (bool)inp->getInput("warning mode", mode_choices, 1);
465 
466  EGS_Input *shape_inp = inp->takeInputItem("shape");
467 
468  EGS_BaseShape *bounds = EGS_BaseShape::createShape(shape_inp);
469 
470  if (!shape_inp) {
471  egsWarning("egs_brachy::checkSourceOverlaps - no `shape` input found.\n");
472  return 1;
473  }
474 
475  EGS_Float bounds_volume = ebvolcor::getShapeVolume(shape_inp);
476  if (bounds_volume < 0){
477  egsWarning("egs_brachy::checkSourceOverlaps - Unable to get shape volume.");
478  return 1;
479  }
480 
481  vector<string> excluded;
482  inp->getInput("excluded geometries", excluded);
483 
484  EGS_Float density;
485  int err = inp->getInput("density of random points (cm^-3)", density);
486  if (err) {
487  egsWarning("egs_brachy::checkSourceOverlaps - The volume correction 'density of random points (cm^-3)' input was not found. Using 1E6/cm^3\n");
488  density = 1E6;
489  }
490  EGS_I64 npoints = (EGS_I64)floor(max(1., density*bounds_volume));
491 
492 
493  EGS_RandomGenerator *rng = EGS_RandomGenerator::defaultRNG();
494  EGS_Vector point;
495 
496  /* find first source geometry defined */
497  EGS_BaseGeometry *base_source = 0;
498  string base_source_name;
499  for (int gg=0; gg < ginfo.ngeom; gg++) {
500  GeomRegionInfo gr = ginfo.ordered_geom_data[gg];
501  if (find(ginfo.source_names.begin(), ginfo.source_names.end(), gr.name) != ginfo.source_names.end()){
502  base_source = EGS_BaseGeometry::getGeometry(gr.name);
503  base_source_name = gr.name;
504  break;
505  }
506  }
507  if (!base_source){
508  egsFatal("egs_brachy::checkSourceOverlaps - did not find base source geometry");
509  }
510 
511  /* now lets see if our base source is in an autoenvelope */
512  EGS_AffineTransform base_transform;
513  EGS_AffineTransform inv_base_transform;
514  for (int gg=0; gg < ginfo.ngeom; gg++) {
515  GeomRegionInfo gr = ginfo.ordered_geom_data[gg];
516  if (find(gr.children.begin(), gr.children.end(), base_source_name) != gr.children.end()){
517  /* base source is child of this geometry */
518  if (gr.type != "EGS_AEnvelope" && gr.type != "EGS_ASwitchedEnvelope"){
519  base_transform = *source_transforms[0];
520  inv_base_transform = base_transform.inverse();
521  }
522  break;
523  }
524  }
525 
526  vector<int> overlaps;
527 
528  for (EGS_I64 i=0; i < npoints; i++) {
529 
530  /* Generate a point and check if its contained within base source. If
531  * it's not actually in a source, go back and generate a new point. */
532  point = bounds->getRandomPoint(rng);
533  base_transform.transform(point);
534  if (base_source->isWhere(point) < 0) {
535  continue;
536  }
537  inv_base_transform.transform(point);
538 
539  for (size_t sa_idx = 0; sa_idx < source_transforms.size(); sa_idx++){
540 
541  overlaps.clear();
542 
543  EGS_Vector transformed(point);
544 
545  // transform from point relative to origin to point
546  // relative to source A we are checking against other sources (B)
547  source_transforms[sa_idx]->transform(transformed);
548 
549  for (size_t sb_idx = sa_idx + 1; sb_idx < source_transforms.size(); sb_idx++){
550 
551  EGS_Vector inner_transformed(transformed);
552 
553  // use current source we are checking transform to back relative to origin
554  source_transforms[sb_idx]->inverse().transform(inner_transformed);
555  base_transform.transform(inner_transformed);
556  if (base_source->isWhere(inner_transformed) >= 0) {
557  /* point falls within Source A & B so they must be overlapping */
558  overlaps.push_back(sa_idx);
559  overlaps.push_back(sb_idx);
560  goto overlap_found;
561  }
562  }
563  }
564  }
565 
566 overlap_found:
567 
568  if (rng){
569  delete rng;
570  }
571  if (shape_inp){
572  delete shape_inp;
573  }
574  timing_blocks.stopTimer();
575 
576  if (overlaps.size() > 1){
577  string msg = "Possible overlap of sources: ";
578  for (int i=0; i < overlaps.size(); i++){
579  msg += to_string(overlaps[i]);
580  if (i != overlaps.size() -1 ){
581  msg += ", ";
582  }
583  }
584  egsInformation((msg+"\n").c_str());
585  }
586 
587  if (overlaps.size() > 1){
588  return fatal ? 1 : 0;
589  }
590 
591  return 0;
592 }
