77 EGS_Float getShapeVolume(EGS_Input *shape_inp) {
78 
79  if (!shape_inp) {
80  return -1;
81  }
82 
83  EGS_Float volume;
84  int err = shape_inp->getInput("shape volume", volume);
85  if (!err) {
86  return volume;
87  }
88 
89  string shape_type;
90  shape_inp->getInput("type", shape_type);
91 
92  if (shape_type == "cylinder") {
93  EGS_Float radius, height;
94  shape_inp->getInput("radius", radius);
95  shape_inp->getInput("height", height);
96  return M_PI*radius*radius*height;
97  } else if (shape_type == "sphere") {
98  EGS_Float radius, height;
99  shape_inp->getInput("radius", radius);
100  shape_inp->getInput("height", height);
101  return 4./3.*M_PI*radius*radius*radius;
102  } else if (shape_type == "spherical shell" || shape_type == "egsSphericalShell") {
103  EGS_Float ri, ro;
104  int hemi;
105  int err = shape_inp->getInput("inner radius", ri);
106  if (err) {
107  egsFatal("getShapeVolume :: missing 'inner radius' input for shape\n");
108  }
109 
110  err = shape_inp->getInput("outer radius", ro);
111  if (err) {
112  egsFatal("getShapeVolume :: missing 'outer radius' input for shape\n");
113  }
114 
115  err = shape_inp->getInput("hemisphere", hemi);
116  if (err) {
117  hemi = 0;
118  }
119  EGS_Float vol = 4./3.*M_PI*(ro*ro*ro - ri*ri*ri);
120  if (hemi !=0) {
121  return vol/2;
122  }
123 
124  return vol;
125 
126  } else if (shape_type == "box") {
127  vector<EGS_Float> box_size;
128  shape_inp->getInput("box size", box_size);
129  if (box_size.size() == 3) {
130  return box_size[0]*box_size[1]*box_size[2];
131  } else {
132  return box_size[0]*box_size[0]*box_size[0];
133  }
134  }
135  if (shape_type == "") {
136  egsWarning("Either include a `type` or `shape volume` input key.");
137  } else {
138  egsWarning(
139  "The volume (in cm^3) for shape type '%s' must be specified using a `shape volume` input key.",
140  shape_type.c_str()
141  );
142  }
143 
144  return -1;
145

305 // source specific MC volume calculation
306 //
307 // Works by generating random points within the sources boundary shape and
308 // then, if the point is within the source, transforming each point to
309 // all the sources locations and checking which phantom region the point lands in.
310 // The volume of a region occupied by a source then is the bounding shape volume
311 // times the fraction of points landing in the source. The volume of the region
312 // then is the uncorrected volume minus the volume occupied by source
313 // Vcor(reg) = V(reg) - V_source(reg)
314 // = V(reg) - V_b * n_points_in_source(reg)/n_total
315 //
316 // where V_b is volume of bounding shape
317 double VolumeCorrector::correctPhantomVolumesForSources() {
318 
319  EGS_Vector point;
320 
321  HitCounterT hit_counter;
322  EGS_I64 n_in_source = 0;
323 
324 
325  for (EGS_I64 i=0; i < source_opts->npoints; i++) {
326 
327  point = source_opts->getRandomPoint();
328  base_transform->transform(point);
329 
330  int ir = base_geom->isWhere(point);
331  if (!ginfo->isSource(ir)) {
332  continue;
333  }
334  n_in_source += 1;
335  base_transform_inv->transform(point);
336 
337 
338  for (size_t sidx = 0; sidx < transforms.size(); sidx++) {
339 
340  EGS_Vector transformed(point);
341  transforms[sidx]->transform(transformed);
342 
343 
344  for (size_t phant= 0; phant<phantoms.size(); phant++) {
345  int reg = phantoms[phant]->geometry->isWhere(transformed);
346  bool in_phant = reg >= 0;
347 
348  if (in_phant) {
349  PhantRegT phant_reg(phant, reg);
350  hit_counter[phant_reg] += 1;
351  }
352 
353  }
354  }
355  }
356 
357 
358  applyVolumeCorrections(source_opts, hit_counter);
359 
360  return source_opts->bounds_volume*(double)n_in_source/(int)source_opts->npoints;

364 void VolumeCorrector::applyVolumeCorrections(Options *opts, HitCounterT hit_counter) {
365 
366  bool zero_dose = opts->mode == ZERO_DOSE;
367  double vol = opts->bounds_volume;
368  double npoints = opts->npoints;
369 
370  for (HitCounterT::iterator hi = hit_counter.begin(); hi != hit_counter.end(); hi++) {
371 
372  int phant_idx = hi->first.first;
373  int phant_reg = hi->first.second;
374  if (phant_reg < 0) {
375  continue;
376  }
377  int hits = hi->second;
378  double reg_vol = phantoms[phant_idx]->getUncorrectedVolume(phant_reg);
379  double corrected_vol = zero_dose ? 0 : max(reg_vol - vol*double(hits)/npoints, 0.);
380  phantoms[phant_idx]->setCorrectedVolume(phant_reg, corrected_vol);
381  }
382 
383 }
